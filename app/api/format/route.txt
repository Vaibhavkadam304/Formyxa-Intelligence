import { NextRequest, NextResponse } from 'next/server';
import { prisma } from '@/lib/prisma';

type PlaceholderSchema = Record<
  string,
  {
    label: string;
    type: string;
  }
>;

// Small helper to fill {{placeholders}} in HTML template
function fillTemplate(htmlTemplate: string, vars: Record<string, any>): string {
  let html = htmlTemplate;
  for (const [key, value] of Object.entries(vars)) {
    html = html.replaceAll(`{{${key}}}`, String(value ?? ''));
  }
  return html;
}

// 🔁 STUB for LLM extraction – replace with real OpenAI call later
function stubExtractVariables(
  rawText: string,
  placeholderSchema: PlaceholderSchema | null,
): Record<string, any> {
  const vars: Record<string, any> = {};

  const keys = placeholderSchema ? Object.keys(placeholderSchema) : [];

  for (const key of keys) {
    if (key === 'executive_summary') {
      // Put full raw text into executive summary for now
      vars[key] = rawText;
    } else if (key === 'client_name') {
      vars[key] = 'Client Name';
    } else if (key === 'scope') {
      vars[key] = 'Scope derived from raw text';
    } else if (key === 'timeline') {
      vars[key] = 'Timeline derived from raw text';
    } else if (key === 'pricing') {
      vars[key] = 'Pricing derived from raw text';
    } else if (key === 'acceptance') {
      vars[key] = 'Acceptance terms derived from raw text';
    } else if (key === 'explanation') {
      vars[key] = rawText; // dump full text here for now
    } else if (key === 'background') {
      vars[key] =
        'Background based on your description (to be generated by AI later).';
    } else if (key === 'supporting_evidence') {
      vars[key] =
        'List of documents you plan to attach (bank statements, bookings, etc.).';
    } else if (key === 'closing') {
      vars[key] =
        'I respectfully request you to kindly reconsider my application in light of the updated information.';
    } else if (key === 'applicant_full_name') {
      vars[key] = 'Vaibhavk';
    } else {
      vars[key] = '';
    }
  }

  // If no schema defined, just use one field
  if (keys.length === 0) {
    vars['body'] = rawText;
  }

  return vars;
}

export const runtime = 'nodejs';

export async function POST(req: NextRequest) {
  try {
    const body = await req.json().catch(() => ({}));
    const { templateSlug, rawText, title } = body as {
      templateSlug?: string;
      rawText?: string;
      title?: string;
    };

    if (!rawText || typeof rawText !== 'string' || !rawText.trim()) {
      return NextResponse.json(
        { error: 'rawText is required' },
        { status: 400 },
      );
    }

    const slug = templateSlug ?? 'website-proposal-standard';

    const template = await prisma.template.findUnique({
      where: { slug },
    });

    if (!template) {
      return NextResponse.json(
        { error: `Template not found for slug "${slug}"` },
        { status: 404 },
      );
    }

    const placeholderSchema =
      (template.placeholderSchema as PlaceholderSchema | null) ?? null;

    // 🔁 In real version: call LLM here with rawText + placeholderSchema
    const variables = stubExtractVariables(rawText, placeholderSchema);

    // ✅ JSON-based path for visa-expiration-letter
    if (slug === 'visa-expiration-letter') {
      const doc = await prisma.document.create({
        data: {
          title: title || template.name || 'Visa expiration letter',
          templateId: template.id,
          variables,
          contentJson: visaExpirationTiptapTemplate as any,
        } as any,
      });

      return NextResponse.json({ id: doc.id }, { status: 201 });
    }

    // 🔁 OLD: HTML template path (website proposal etc.)
    if (!template.htmlTemplate) {
      return NextResponse.json(
        {
          error: 'Template has no htmlTemplate (and is not JSON-based)',
        },
        { status: 500 },
      );
    }

    const contentHtml = fillTemplate(template.htmlTemplate, variables);

    const doc = await prisma.document.create({
      data: {
        title: title || 'Untitled document',
        templateId: template.id,
        variables,
        contentHtml,
      },
    });

    return NextResponse.json({ id: doc.id }, { status: 201 });
  } catch (err) {
    console.error(err);
    return NextResponse.json({ error: 'Server error' }, { status: 500 });
  }
}
