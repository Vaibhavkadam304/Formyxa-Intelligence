"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";
import {
  FileText,
  Send,
  CheckCircle2,
  X,
  Shield,
} from "lucide-react";

import { Button } from "@/components/ui/button";
import { Card, CardContent } from "@/components/ui/card";
import { Tabs, TabsList, TabsTrigger } from "@/components/ui/tabs";
import { Badge } from "@/components/ui/badge";
import Header from "@/components/header";
import Footer from "@/components/footer";

type DesignKey =
  | "offer_modern_blue"
  | "offer_minimal_plain"
  | "offer_classic_border"
  | "offer_green_wave";

interface DesignOption {
  id: DesignKey;
  name: string;
  description: string;
  imageUrl: string;
}

const DESIGNS: DesignOption[] = [
  {
    id: "offer_modern_blue",
    name: "Modern blue header",
    description: "Blue header band with HR signature block.",
    imageUrl: "/previews/offer-modern-blue.jpg",
  },
  {
    id: "offer_minimal_plain",
    name: "Minimal plain",
    description: "Clean text-only format, company standard.",
    imageUrl: "/previews/offer-modern-pale.jpg",
  },
  {
    id: "offer_classic_border",
    name: "Classic border",
    description: "Formal bordered layout used in enterprises.",
    imageUrl: "/previews/offer-classic-border.jpg",
  },
  {
    id: "offer_green_wave",
    name: "Fresh green wave",
    description: "Modern layout with subtle green header.",
    imageUrl: "/previews/offer-modern-green.jpg",
  },
];


const DOC_TO_TEMPLATE_SLUG: Record<string, string> = {
  "offer-letter": "offer-letter-standard",
  "appointment-letter": "appointment-letter-standard",
  "resignation-letter": "resignation-letter-standard",
};

export default function HROfferWorkspacePage() {
  const router = useRouter();
  const [activeDoc, setActiveDoc] = useState("offer-letter");
  const [preset, setPreset] = useState("corporate");
  const [selectedDesign, setSelectedDesign] =
    useState<DesignKey | null>(null);

  const selectedDesignData = DESIGNS.find(
    (d) => d.id === selectedDesign,
  );

  const handleContinue = () => {
    if (!selectedDesign) return;

    const params = new URLSearchParams({
      template: DOC_TO_TEMPLATE_SLUG[activeDoc],
      preset,
      design: selectedDesign,
    });

    router.push(`/new?${params.toString()}`);
  };

  return (
    <div className="min-h-screen bg-[#F8FAFC] flex flex-col">
      <Header />

      <main className="max-w-7xl mx-auto w-full py-10 px-6 flex-1">
        <div className="flex gap-10 items-start">
          {/* Sidebar */}
          <aside className="w-64 sticky top-24">
            <Card className="border border-slate-200 shadow-sm">
              <CardContent className="p-5">
                <h3 className="text-sm font-semibold text-slate-900 mb-1">
                  HR & People Operations
                </h3>
                <Badge className="mb-5 bg-blue-50 text-blue-600 border-blue-100 text-[10px] uppercase font-semibold">
                  HR Documents
                </Badge>

                <nav className="space-y-1">
                  {[
                    { id: "offer-letter", label: "Offer letter", icon: Send },
                    {
                      id: "appointment-letter",
                      label: "Appointment letter",
                      icon: FileText,
                    },
                    {
                      id: "resignation-letter",
                      label: "Resignation letter",
                      icon: FileText,
                    },
                  ].map((item) => (
                    <button
                      key={item.id}
                      onClick={() => setActiveDoc(item.id)}
                      className={`w-full flex items-center gap-3 px-3 py-2.5 rounded-lg text-sm font-medium transition ${
                        activeDoc === item.id
                          ? "bg-blue-50 text-blue-600"
                          : "text-slate-600 hover:bg-slate-50"
                      }`}
                    >
                      <item.icon className="w-4 h-4" />
                      {item.label}
                    </button>
                  ))}
                </nav>
              </CardContent>
            </Card>
          </aside>

          {/* Main */}
          <div className="flex-1">
            <div className="mb-8">
              <div className="text-xs font-semibold uppercase tracking-widest text-slate-400">
                Official HR Documents
              </div>
              <h1 className="text-[28px] font-semibold tracking-tight text-slate-900 mt-1">
                HR & People Operations
              </h1>
              <p className="text-sm text-slate-500 max-w-xl mt-1">
                Define your company’s official offer letter — structure first, design second.
              </p>
            </div>

            {/* Presets */}
            <div className="mb-8">
              <Tabs value={preset} onValueChange={setPreset}>
                <TabsList
                  className="
                    inline-flex items-center gap-1
                    rounded-full bg-slate-100
                    p-1
                    max-w-xl
                  "
                >
                  <TabsTrigger
                    value="startup"
                    className="
                      rounded-full px-6 py-2 text-sm font-medium
                      text-slate-600
                      data-[state=active]:bg-white
                      data-[state=active]:text-slate-900
                      data-[state=active]:shadow-sm
                    "
                  >
                    Startup
                  </TabsTrigger>

                  <TabsTrigger
                    value="corporate"
                    className="
                      rounded-full px-6 py-2 text-sm font-medium
                      text-slate-600
                      data-[state=active]:bg-white
                      data-[state=active]:text-slate-900
                      data-[state=active]:shadow-sm
                    "
                  >
                    Corporate
                  </TabsTrigger>

                  <TabsTrigger
                    value="custom"
                    className="
                      rounded-full px-6 py-2 text-sm font-medium
                      text-slate-600
                      data-[state=active]:bg-white
                      data-[state=active]:text-slate-900
                      data-[state=active]:shadow-sm
                    "
                  >
                    Custom / Legacy
                  </TabsTrigger>
                </TabsList>

              </Tabs>
              <p className="text-[11px] text-slate-400 mt-2">
                Presets define wording and sections used across all future offer letters.
              </p>
            </div>

            <div className={`flex gap-8 transition-all ${selectedDesign ? "items-start" : ""}`}>
              {/* Designs */}
              <div className={`${selectedDesign ? "w-[45%]" : "w-full"} transition-all`}>
                <div className="flex items-center justify-between mb-6">
                  <h3 className="text-sm font-semibold text-slate-900">
                    Choose company-standard format
                  </h3>
                  <div className="flex items-center gap-1 text-xs text-slate-400">
                    <Shield className="w-3.5 h-3.5" />
                    Company standard
                  </div>
                </div>

                <div
                  className={`grid gap-4 ${
                    selectedDesign ? "grid-cols-1" : "grid-cols-2"
                  }`}
                >
                  {DESIGNS.map((design) => (
                    <Card
                      key={design.id}
                      onClick={() => setSelectedDesign(design.id)}
                      className={`
                        cursor-pointer
                        rounded-2xl
                        border
                        bg-white
                        transition
                        ${
                          selectedDesign === design.id
                            ? "border-blue-400 bg-blue-50/30"
                            : "border-slate-200 hover:border-slate-300"
                        }
                      `}
                    >
                      <CardContent className="p-0">
                        {/* Card header */}
                        <div className="px-5 pt-4 pb-3">
                          <div className="flex items-start justify-between">
                            <div>
                              <h4 className="text-sm font-semibold text-slate-900">
                                {design.name}
                              </h4>
                              <p className="mt-0.5 text-xs text-slate-400 leading-snug">
                                {design.description}
                              </p>
                            </div>

                            {selectedDesign === design.id && (
                              <CheckCircle2 className="h-5 w-5 text-blue-600 mt-0.5" />
                            )}
                          </div>
                        </div>

                        {/* Preview area */}
                        <div className="px-5 pb-5">
                          <div className="h-full w-full rounded-lg bg-white shadow-sm p-4">
                            <img
                              src={design.imageUrl}
                              alt={design.name}
                              className="h-full w-full object-cover opacity-80"
                            />
                          </div>
                        </div>
                      </CardContent>

                    </Card>
                  ))}
                </div>

                {/* CTA */}
                <div className="mt-10 flex items-center justify-between rounded-lg border border-slate-200 bg-slate-50 px-5 py-4">
                  <Button
                    onClick={handleContinue}
                    disabled={!selectedDesign}
                    className="h-11 px-8 rounded-md text-sm font-semibold"
                  >
                    Use this design
                  </Button>
                  <div className="text-xs text-slate-500">
                    {selectedDesign ? (
                      <>
                        Selected:{" "}
                        <span className="font-medium text-slate-900">
                          {selectedDesignData?.name}
                        </span>
                        <br />
                        Next: Candidate details
                      </>
                    ) : (
                      "Select a design to continue"
                    )}
                  </div>
                </div>
              </div>

              {/* Preview */}
             {selectedDesign && (
                <div className="w-[55%] sticky top-24">
                  <div className="bg-white rounded-2xl shadow-xl p-6">
                    <img
                      src={selectedDesignData?.imageUrl}
                      alt="Company standard document preview"
                      className="w-full rounded-lg"
                    />
                  </div>
                </div>
              )}
            </div>
          </div>
        </div>
      </main>

      <Footer />
    </div>
  );
}
