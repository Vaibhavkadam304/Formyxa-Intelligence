"use client"

import { useEffect, useState, ReactNode, useRef } from "react"
import { useRouter, useSearchParams } from "next/navigation"
import { Check } from "lucide-react"

import Header from "@/components/header"
import Footer from "@/components/footer"
import type { TemplateConfig } from "@/lib/useCases"
import {
  TEMPLATE_CATEGORIES,
  getCategoryById,
  getTemplatesForCategory,
} from "@/lib/useCases"
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogDescription,
} from "@/components/ui/dialog"

type CategoryColor =
  | "blue"
  | "amber"
  | "indigo"
  | "purple"
  | "slate"
  | "orange"
  | "teal"
  | "pink"

type CategoryCardProps = {
  icon: ReactNode
  title: string
  description: string
  color: CategoryColor
  isSelected: boolean
  onClick: () => void
}

// If you add `previewImage` to TemplateConfig, this will match it.
type TemplateWithImage = TemplateConfig & {
  previewImage?: string
}

/* ----------------- Offer design variants ----------------- */

type OfferDesignKey =
  | "offer_modern_blue"
  | "offer_minimal_plain"
  | "offer_classic_border"
  | "offer_green_wave"

// Website proposal modes (quick email vs full doc)
type WebsiteProposalMode = "quick-email" | "full-doc"

const OFFER_DESIGNS: {
  key: OfferDesignKey
  label: string
  description: string
  preview?: string
}[] = [
  {
    key: "offer_modern_blue",
    label: "Modern blue header",
    description: "Blue header band with HR signature block.",
    preview: "/previews/offer-modern-blue.png",
  },
  {
    key: "offer_minimal_plain",
    label: "Minimal plain",
    description: "No header image, just clean text + signature.",
    preview: "/previews/offer-minimal-plain.png",
  },
  {
    key: "offer_classic_border",
    label: "Classic border",
    description: "Thin border, serif headings, very formal.",
    preview: "/previews/offer-classic-border.png",
  },
  {
    key: "offer_green_wave",
    label: "Fresh green wave",
    description: "Green wave header with clean, friendly layout.",
    preview: "/previews/offer-green-wave.png",
  },
]

export default function ChoosePurposePage() {
  const router = useRouter()
  const searchParams = useSearchParams()

  const [selectedCategoryId, setSelectedCategoryId] = useState<string | null>(
    null,
  )

  // Which template currently has the website proposal mode picker open
  const [openWebsiteProposalSlug, setOpenWebsiteProposalSlug] = useState<
    string | null
  >(null)

  // Which template currently has design picker open (offer letter)
  const [openDesignForSlug, setOpenDesignForSlug] = useState<string | null>(
    null,
  )

  const selectedCategory =
    selectedCategoryId !== null
      ? getCategoryById(selectedCategoryId)
      : undefined
  const templates = (selectedCategoryId
    ? getTemplatesForCategory(selectedCategoryId)
    : []) as TemplateWithImage[]

  const railRef = useRef<HTMLDivElement | null>(null)

  // If /choose?category=xyz, open that category directly
  useEffect(() => {
    const fromQuery = searchParams.get("category")
    if (fromQuery) {
      const exists = getCategoryById(fromQuery)
      if (exists) setSelectedCategoryId(fromQuery)
    }
  }, [searchParams])

  const handleUseTemplate = (tpl: TemplateConfig) => {
    // Special handling: Job offer letter → open design picker instead of routing
    if (tpl.slug === "offer-letter-standard") {
      setOpenDesignForSlug((prev) => (prev === tpl.slug ? null : tpl.slug))
      // close other special UIs
      setOpenWebsiteProposalSlug(null)
      return
    }

    // Special handling: Website project proposal → open mode picker (quick email vs full doc)
    if (tpl.slug === "website-project-proposal") {
      setOpenWebsiteProposalSlug((prev) => (prev === tpl.slug ? null : tpl.slug))
      // close other special UIs
      setOpenDesignForSlug(null)
      return
    }

    // Normal behaviour for all other templates
    setSelectedCategoryId(null)
    setOpenDesignForSlug(null)
    setOpenWebsiteProposalSlug(null)

    if (tpl.customRoute) {
      router.push(tpl.customRoute)
    } else {
      router.push(`/new?template=${encodeURIComponent(tpl.slug)}`)
    }
  }

  // When user actually chooses a design for the offer letter
  const handleOfferDesignSelect = (design: OfferDesignKey) => {
    setOpenDesignForSlug(null)
    setSelectedCategoryId(null)

    router.push(
      `/new?template=${encodeURIComponent(
        "offer-letter-standard",
      )}&design=${encodeURIComponent(design)}`,
    )
  }

  const handleWebsiteProposalModeSelect = (mode: WebsiteProposalMode) => {
    setOpenWebsiteProposalSlug(null)
    setSelectedCategoryId(null)
    setOpenDesignForSlug(null)

    if (mode === "quick-email") {
      router.push(
        `/new?template=${encodeURIComponent(
          "website-project-proposal-quick-email",
        )}`,
      )
    } else {
      // Full professional DOCX proposal uses the existing template
      router.push(
        `/new?template=${encodeURIComponent("website-project-proposal")}`,
      )
    }
  }

  const colorCycle: CategoryColor[] = [
    "blue",
    "amber",
    "indigo",
    "purple",
    "slate",
    "orange",
    "teal",
    "pink",
  ]

  return (
    <div className="relative min-h-screen bg-gradient-to-b from-background via-background to-muted">
      {/* soft glow */}
      <div className="pointer-events-none absolute inset-x-0 -top-32 -z-10 h-64 bg-[radial-gradient(circle_at_top,_rgba(129,140,248,0.3),_transparent_60%)]" />

      <Header />

      <main className="max-w-6xl mx-auto pt-28 pb-20 px-4 md:px-6 space-y-10">
        {/* Heading */}
        <section className="flex flex-col items-center text-center space-y-3 animate-in fade-in-0 slide-in-from-bottom-4 duration-700">
          <h1 className="text-3xl md:text-4xl font-semibold tracking-tight text-foreground text-balance">
            Choose what you need to write
          </h1>
          <p className="text-xs md:text-sm text-muted-foreground max-w-xl">
            Pick a card to start with the right template. You’ll fill in your
            story on the next screen.
          </p>
        </section>

        {/* Dashboard-style category grid */}
        <section className="relative">
          <div className="relative rounded-[32px] bg-white/95 shadow-[0_22px_70px_rgba(15,23,42,0.18)] px-4 sm:px-6 md:px-8 py-6 md:py-8">
            {/* MOBILE rail */}
            <div
              ref={railRef}
              className="
                mt-1 flex gap-5
                overflow-x-auto
                pb-2
                snap-x snap-mandatory
                scrollbar-none
                md:hidden
              "
            >
              {TEMPLATE_CATEGORIES.map((category, index) => {
                const color = colorCycle[index % colorCycle.length]
                const isSelected = selectedCategoryId === category.id

                return (
                  <div
                    key={category.id}
                    className="snap-center shrink-0 w-[230px]"
                  >
                    <PurposeCard
                      icon={<span className="text-3xl">{category.icon}</span>}
                      title={category.label}
                      description={category.description}
                      color={color}
                      isSelected={isSelected}
                      onClick={() => setSelectedCategoryId(category.id)}
                    />
                  </div>
                )
              })}
            </div>

            {/* DESKTOP grid */}
            <div
              className="
                mt-1 hidden
                md:grid md:grid-cols-3 xl:grid-cols-4
                gap-6
              "
            >
              {TEMPLATE_CATEGORIES.map((category, index) => {
                const color = colorCycle[index % colorCycle.length]
                const isSelected = selectedCategoryId === category.id

                return (
                  <PurposeCard
                    key={category.id}
                    icon={<span className="text-3xl">{category.icon}</span>}
                    title={category.label}
                    description={category.description}
                    color={color}
                    isSelected={isSelected}
                    onClick={() => setSelectedCategoryId(category.id)}
                  />
                )
              })}
            </div>
          </div>
        </section>
      </main>

      {/* Template chooser modal */}
      <Dialog
        open={selectedCategoryId !== null}
        onOpenChange={(open) => {
          if (!open) {
            setSelectedCategoryId(null)
            setOpenDesignForSlug(null)
            setOpenWebsiteProposalSlug(null)
          }
        }}
      >
        <DialogContent
          className="
            max-w-5xl w-full border-0 bg-transparent p-0
            sm:max-w-4xl
          "
        >
          {selectedCategory && (
            <div
              className="
                relative w-full overflow-hidden
                rounded-[28px] bg-white
                shadow-[0_22px_70px_rgba(15,23,42,0.18)]
                max-h-[80vh] flex flex-col
              "
            >
              {/* header inside the panel */}
              <div className="px-6 pt-6 pb-3 border-b border-slate-100">
                <DialogHeader className="space-y-2 text-left">
                  <DialogTitle className="flex items-center gap-2 text-lg font-semibold">
                    <span className="text-xl">{selectedCategory.icon}</span>
                    <span>{selectedCategory.label}</span>
                  </DialogTitle>
                  <DialogDescription className="text-xs sm:text-sm text-muted-foreground">
                    Choose a template for this purpose. You&apos;ll describe
                    your situation on the next screen and the AI will fill the
                    placeholders for you.
                  </DialogDescription>
                </DialogHeader>
              </div>

              {/* scrollable grid area */}
              <div className="px-4 sm:px-6 pb-5 pt-4 overflow-y-auto no-scrollbar">
                <div className="grid gap-4 md:grid-cols-2">
                  {templates.map((tpl) => {
                    const showDesignPicker =
                      tpl.slug === "offer-letter-standard" &&
                      openDesignForSlug === tpl.slug

                    const showWebsiteModePicker =
                      tpl.slug === "website-project-proposal" &&
                      openWebsiteProposalSlug === tpl.slug

                    return (
                      <div key={tpl.slug} className="space-y-2">
                        {/* main card */}
                        <button
                          type="button"
                          onClick={() => handleUseTemplate(tpl)}
                          className="
                            group flex flex-col sm:flex-row sm:items-stretch gap-4
                            rounded-3xl border border-slate-100
                            bg-slate-50/60
                            p-4 sm:p-5
                            text-left
                            shadow-[0_8px_28px_rgba(15,23,42,0.06)]
                            hover:-translate-y-1
                            hover:border-indigo-200
                            hover:bg-white
                            hover:shadow-[0_14px_40px_rgba(15,23,42,0.10)]
                            transition-all duration-200
                          "
                        >
                          {/* preview image */}
                          <div className="w-full sm:w-40 md:w-44 flex-shrink-0 sm:self-stretch">
                            <div
                              className="
                                relative
                                h-28 sm:h-full sm:min-h-[150px]
                                w-full
                                rounded-2xl overflow-hidden
                                bg-slate-100
                              "
                            >
                              {tpl.previewImage ? (
                                <img
                                  src={tpl.previewImage}
                                  alt={tpl.title}
                                  className="absolute inset-0 h-full w-full object-cover"
                                />
                              ) : (
                                <div className="absolute inset-0 bg-gradient-to-br from-slate-50 via-slate-100 to-slate-200" />
                              )}
                            </div>
                          </div>

                          {/* right side content */}
                          <div className="flex flex-1 flex-col justify-between gap-2">
                            <div className="space-y-1">
                              <p className="text-sm font-semibold text-foreground">
                                {tpl.title}
                              </p>
                              <p className="text-xs text-muted-foreground line-clamp-2">
                                {tpl.description}
                              </p>
                              {tpl.examplePrompt && (
                                <p className="mt-1 text-[11px] text-slate-500 line-clamp-2">
                                  <span className="font-medium text-slate-700">
                                    Example:{" "}
                                  </span>
                                  {tpl.examplePrompt}
                                </p>
                              )}
                            </div>

                            <div className="mt-2 flex items-center justify-between">
                              <span className="text-[11px] text-slate-500">
                                Click anywhere on this card to use this
                                template.
                              </span>
                              <span
                                className="
                                  inline-flex items-center justify-center
                                  rounded-full px-3 py-1
                                  text-[11px] font-medium
                                  bg-indigo-500 text-white
                                  shadow-sm group-hover:bg-indigo-600
                                "
                              >
                                {tpl.slug === "offer-letter-standard"
                                  ? "Choose design"
                                  : "Use template"}
                              </span>
                            </div>
                          </div>
                        </button>

                        {/* design picker only for job offer */}
                        {showDesignPicker && (
                          <OfferDesignPicker
                            onSelect={handleOfferDesignSelect}
                          />
                        )}

                        {/* website proposal mode picker */}
                        {showWebsiteModePicker && (
                          <WebsiteProposalModePicker
                            onSelect={handleWebsiteProposalModeSelect}
                          />
                        )}
                      </div>
                    )
                  })}

                  {templates.length === 0 && (
                    <p className="text-xs text-muted-foreground">
                      No templates configured yet for this purpose. You can add
                      some in <code>lib/useCases.ts</code>.
                    </p>
                  )}
                </div>
              </div>
            </div>
          )}
        </DialogContent>
      </Dialog>

      <Footer />
    </div>
  )
}

/* ---- Design picker component ---- */

function OfferDesignPicker({
  onSelect,
}: {
  onSelect: (design: OfferDesignKey) => void
}) {
  return (
    <div className="mt-1 rounded-2xl border border-indigo-100 bg-indigo-50/70 p-3">
      <p className="mb-2 text-[11px] font-medium text-indigo-700">
        Choose a design for this job offer letter
      </p>
      <div className="grid gap-2 sm:grid-cols-3">
        {OFFER_DESIGNS.map((design) => (
          <button
            key={design.key}
            type="button"
            onClick={() => onSelect(design.key)}
            className="
              group flex flex-col gap-1
              rounded-xl border border-indigo-100 bg-white p-2 text-left
              hover:border-indigo-400 hover:shadow-sm
              transition-all
            "
          >
            <div className="mb-1 h-14 rounded-lg bg-slate-100 overflow-hidden">
              {design.preview ? (
                <img
                  src={design.preview}
                  alt={design.label}
                  className="h-full w-full object-cover"
                />
              ) : (
                <div className="flex h-full items-center justify-center text-[10px] text-slate-400">
                  Preview
                </div>
              )}
            </div>
            <span className="text-[11px] font-semibold text-slate-800">
              {design.label}
            </span>
            <span className="text-[10px] text-slate-500">
              {design.description}
            </span>
          </button>
        ))}
      </div>
    </div>
  )
}

function WebsiteProposalModePicker({
  onSelect,
}: {
  onSelect: (mode: WebsiteProposalMode) => void
}) {
  return (
    <div className="mt-1 rounded-2xl border border-blue-100 bg-blue-50/70 p-3">
      <p className="mb-2 text-[11px] font-medium text-blue-700">
        Choose proposal type
      </p>
      <div className="grid gap-2 sm:grid-cols-2">
        {/* Quick email */}
        <button
          type="button"
          onClick={() => onSelect("quick-email")}
          className="
            group flex flex-col gap-1
            rounded-xl border border-slate-200 bg-white p-2 text-left
            hover:border-blue-400 hover:shadow-sm
            transition-all
          "
        >
          <span className="text-[11px] font-semibold text-slate-900">
            Quick proposal email
          </span>
          <span className="text-[10px] text-slate-500">
            Short email for fast replies on WhatsApp, Upwork, or email.
          </span>
        </button>

        {/* Full document */}
        <button
          type="button"
          onClick={() => onSelect("full-doc")}
          className="
            group flex flex-col gap-1
            rounded-xl border border-slate-200 bg-white p-2 text-left
            hover:border-blue-400 hover:shadow-sm
            transition-all
          "
        >
          <span className="text-[11px] font-semibold text-slate-900">
            Full professional proposal (DOCX)
          </span>
          <span className="text-[10px] text-slate-500">
            Detailed multi-section proposal with scope, pricing and timeline.
          </span>
        </button>
      </div>
    </div>
  )
}

/* ---- Category card ---- */

function PurposeCard({
  icon,
  title,
  description,
  color,
  isSelected,
  onClick,
}: CategoryCardProps) {
  const styles: Record<
    CategoryColor,
    { artboard: string; selected: string }
  > = {
    blue: {
      artboard:
        "bg-gradient-to-br from-blue-50 via-blue-100 to-blue-200",
      selected: "ring-2 ring-blue-400 shadow-xl shadow-blue-100",
    },
    amber: {
      artboard:
        "bg-gradient-to-br from-amber-50 via-amber-100 to-amber-200",
      selected: "ring-2 ring-amber-400 shadow-xl shadow-amber-100",
    },
    indigo: {
      artboard:
        "bg-gradient-to-br from-indigo-50 via-indigo-100 to-indigo-200",
      selected: "ring-2 ring-indigo-400 shadow-xl shadow-indigo-100",
    },
    purple: {
      artboard:
        "bg-gradient-to-br from-purple-50 via-purple-100 to-purple-200",
      selected: "ring-2 ring-purple-400 shadow-xl shadow-purple-100",
    },
    slate: {
      artboard:
        "bg-gradient-to-br from-slate-50 via-slate-100 to-slate-200",
      selected: "ring-2 ring-slate-400 shadow-xl shadow-slate-100",
    },
    orange: {
      artboard:
        "bg-gradient-to-br from-orange-50 via-orange-100 to-orange-200",
      selected: "ring-2 ring-orange-400 shadow-xl shadow-orange-100",
    },
    teal: {
      artboard:
        "bg-gradient-to-br from-teal-50 via-teal-100 to-teal-200",
      selected: "ring-2 ring-teal-400 shadow-xl shadow-teal-100",
    },
    pink: {
      artboard:
        "bg-gradient-to-br from-pink-50 via-pink-100 to-pink-200",
      selected: "ring-2 ring-pink-400 shadow-xl shadow-pink-100",
    },
  }

  const { artboard, selected } = styles[color]

  return (
    <button
      type="button"
      onClick={onClick}
      className={`
        group relative flex h-[260px] flex-col
        items-start justify-between
        rounded-3xl border
        bg-white/95
        px-5 pt-5 pb-4 text-left
        shadow-[0_8px_30px_rgba(15,23,42,0.06)]
        transition-all duration-300
        hover:-translate-y-1.5
        hover:shadow-[0_18px_45px_rgba(15,23,42,0.12)]
        hover:border-indigo-200
        ${isSelected ? selected : "border-slate-100"}
      `}
    >
      {isSelected && (
        <div className="absolute -right-2 -top-2 flex h-7 w-7 items-center justify-center rounded-full bg-emerald-500 shadow-md">
          <Check className="h-4 w-4 text-white" />
        </div>
      )}

      {/* “image” area */}
      <div
        className={`
          w-full rounded-2xl ${artboard}
          flex items-center justify-center
          border border-white/70
          shadow-[0_6px_18px_rgba(148,163,184,0.25)]
          transition-transform duration-300
          group-hover:translate-y-[-3px]
        `}
        style={{ height: 132 }}
      >
        <div
          className="
            h-14 w-14 rounded-2xl bg-white
            shadow-[0_8px_24px_rgba(15,23,42,0.20)]
            flex items-center justify-center text-2xl
            transition-transform duration-300
            group-hover:scale-105
          "
        >
          {icon}
        </div>
      </div>

      {/* text area */}
      <div className="mt-3 w-full space-y-1">
        <h3 className="text-sm font-semibold text-foreground">{title}</h3>
        <p className="text-xs text-muted-foreground leading-relaxed line-clamp-2">
          {description}
        </p>
      </div>

      {/* grey skeleton lines */}
      <div className="mt-3 flex w-full flex-col gap-1.5">
        <div className="h-1.5 w-5/6 rounded-full bg-slate-100" />
        <div className="h-1.5 w-2/3 rounded-full bg-slate-100" />
      </div>
    </button>
  )
}
